\begin{algorithm}
\caption{Additional Action during Page Mapping}
\label{alg:page-mapping}
\begin{algorithmic}[1]
\STATE Let $S$ be the group size
\STATE Let $bitvec$ be a temporary bit vector, initially all zero
\STATE \textbf{Input:} regular $ppn$, $page.use\_tracker$, $cgroup\_id$
\STATE bitvec.init($S$)
\FOR{i = 0 to use\_tracker.size() - 1}
    \STATE bitvec $|=$ (1 \url{<<} use\_tracker[i].group\_member\_id)
    \IF{use\_tracker[i].cgroup\_id == cgroup\_id}
        \STATE use\_tracker[i].map\_count += 1
        \STATE \textbf{goto} FINALIZE
    \ENDIF
\ENDFOR
\STATE group\_member\_id = pick\_rand\_0\_position\_from(bitvec)
\STATE use\_tracker.insert(cgroup\_id, group\_member\_id, map\_count = 1)
\STATE \textbf{FINALIZE:}
\IF{use\_tracker.size() $>$ 1}
    \STATE ppn' = N + ppn * S + group\_member\_id \hfill \textcolor{blue} {// $N$ is the total number of regular PPNs, See Equation \ref{eq:sharingonlyppn}}
    \STATE install\_pte(ppn')
\ELSE
    \STATE install\_pte(ppn) \hfill // Regular PPN
    \STATE write\_back($CL(ppn)s$)
\ENDIF
\end{algorithmic}
\end{algorithm}
