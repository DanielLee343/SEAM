\begin{algorithm}
\caption{Additional Action during Page Unmapping}
\label{alg:page-unmapping}
\begin{algorithmic}[1]
\STATE \textbf{Input:} $sharing\_only\_ppn$, $cgroup\_id$
\STATE flush($CL(sharing\_only\_ppn)s$)
\STATE ppn = (sharing\_only\_ppn - N ) / S \hfill \textcolor{blue} {// See Equation \ref{eq:regularppn}}
\STATE use\_tracker = pfn\_to\_page(ppn).use\_tracker
\FOR{i = 0 to use\_tracker.size() - 1}
    \IF{use\_tracker[i].cgroup\_id == cgroup\_id}
        % \STATE clear\_pte(ppn)
        \STATE use\_tracker[i].map\_count -= 1
        \IF{use\_tracker[i].map\_count == 0}
            \STATE DELETE use\_tracker[i]
        \ENDIF
        \STATE return
    \ENDIF
\ENDFOR

\end{algorithmic}
\end{algorithm}
